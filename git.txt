========================
GIT COMMANDS ‚Äì COMPLETE INTERVIEW NOTES
========================

# Git Configuration
git config --global user.name "Sarvesh"
git config --global user.email "123@jio.com"

# Check configuration
git config --global user.name
git config --global user.email

# Edit git config file directly
git config --global --edit


# Initialize Repository
git init


# Check status of repository
git status


# Add files (Working Directory ‚Üí Staging Area)
git add .
git add <filename>


# Commit changes (Staging Area ‚Üí Repository)
git commit -m "commit message"


# View commit history
git log
git log --oneline


# Go to previous commit (Detached HEAD)
git checkout <commit-hash>

# Return to main branch
git checkout main
# (or git checkout master)


# Branch commands
git branch
git branch <branch-name>
git checkout <branch-name>
git checkout -b <branch-name>


# Merge branches
git merge <branch-name>
git merge --abort        # Abort merge if conflict occurs


# Rebase (Linear history)
git rebase main
git rebase -i HEAD~3
git rebase --abort


# Cherry-pick (pick specific commit)
git cherry-pick <commit-id>


# Remote repository (GitHub)
git remote add origin <repo-url>
git remote -v


# Push code to GitHub
git push -u origin main


# Fetch and Pull
git fetch
git pull origin main


# Undo changes
git checkout -- <file>          # Undo file changes before staging
git reset <file>                # Unstage file
git reset --soft HEAD~1         # Undo last commit (keep changes)
git reset --hard HEAD~1         # Undo last commit (delete changes)


# Restore commands (newer Git)
git restore <file>
git restore --staged <file>


# Stash commands
git stash
git stash list
git stash apply
git stash pop


# Delete files/folders
git rm <file>
git rm -r <folder>


# Clean untracked files/folders
git clean -f
git clean -fd


# Tags (Release management)
git tag
git tag v1.0
git push origin v1.0
git push origin --tags


# Submodules
git submodule add <repo-url>
git submodule update --init --recursive
git submodule status


# Check differences
git diff
git diff --staged


# Show file commit history
git blame <file>


# Show commit details
git show


# Git Flow
Working Directory ‚Üí Staging Area ‚Üí Repository

# Important Concepts
HEAD  ‚Üí Points to current commit
Branch ‚Üí Pointer to commits
Remote ‚Üí GitHub/GitLab repository
